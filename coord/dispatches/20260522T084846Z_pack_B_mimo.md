@budget.command('summary')
@click.option('--since-iso', default=None, help='ISO timestamp cutoff')
@click.option('--since-days', type=int, default=None, help='Number of days to look back (min 1). Mutually exclusive with --since-iso.')
def budget_summary(since_iso, since_days):
    if since_iso is not None and since_days is not None:
        raise click.UsageError('--since-iso and --since-days are mutually exclusive.')
    if since_days is not None:
        if since_days < 1:
            raise click.BadParameter('--since-days must be at least 1', param_hint='--since-days')
        from datetime import datetime, timedelta
        since_iso = (datetime.utcnow() - timedelta(days=since_days)).isoformat() + 'Z'
    """Show budget summary."""
